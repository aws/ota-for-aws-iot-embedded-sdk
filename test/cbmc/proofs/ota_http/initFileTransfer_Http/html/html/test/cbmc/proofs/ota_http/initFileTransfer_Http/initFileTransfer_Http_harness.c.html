30 /* Stub required for the proof. */
31 OtaHttpStatus_t init( char * pUrl )
32 {
33 OtaHttpStatus_t status;
34
35 return status;
36 }
37
38 void initFileTransfer_Http_harness()
39 {
40 OtaAgentContext_t * pAgentCtx;
41 OtaFileContext_t fileContext;
42 OtaHttpInterface_t http;
43 OtaHttpStatus_t status;
44 OtaInterfaces_t interface;
45 OtaAgentContext_t agent;
46
47 pAgentCtx = &agent;
48
49 /* Initialize the file context from the Agent context. */
50 pAgentCtx->fileContext = fileContext;
51
52 /* Updating the function pointer in Http to the stub. */
53 http.init = init;
54
55 /* Update the interface and the Agent. */
56 interface.http = http;
57 pAgentCtx->pOtaInterface = &interface;
58
59 status = initFileTransfer_Http( pAgentCtx );
60
61 /* The function should either return OtaErrNone or OtaErrCleanupDataFailed. */
62 __CPROVER_assert( ( status == OtaErrNone ) ||
63 ( status == OtaErrInitFileTransferFailed ),
64 "The function return should be either OtaErrNone or OtaErrCleanupDataFailed" );
65 }
