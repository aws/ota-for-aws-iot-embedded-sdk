23 /**
24 * @file setDataInterface_harness.c
25 * @brief Implements the proof harness for setDataInterface function.
26 */
27 /* Ota interface includes. */
28 #include "ota_interface_private.h"
29
30 void setDataInterface_harness()
31 {
32 OtaDataInterface_t * pDataInterface;
33 OtaDataInterface_t dataInterface;
34 OtaErr_t err;
35
36 uint8_t * pProtocol;
37 size_t size;
38
39 /* Initialize the variable. */
40 pProtocol = ( uint8_t * ) malloc( size );
41 pDataInterface = &dataInterface;
42
43 /* Call the function under test. */
44 err = setDataInterface( pDataInterface, pProtocol );
45
46 /* The function return can only be either of two values i.e OtaErrNone or
47 * OtaErrInvalidDataProtocol. */
48 __CPROVER_assert( ( err == OtaErrInvalidDataProtocol ) || ( err == OtaErrNone ),
49 "The function return can be either of those values." );
50
51 /* Free variables. */
52 free( pProtocol );
53 }
