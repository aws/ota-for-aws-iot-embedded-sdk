35 /**
36 * @otamqttpage
37 * @brief The OTA PAL interface definition.
38 *
39 * @otamqttsectionoverview
40 *
41 * The OTA MQTT interface is a set of APIs that must be implemented by
42 * a library to enable the OTA library to connect to AWS IoT and manage
43 * notification and request data. The OTA library uses MQTT PUBLISH
44 * messages to inform AWS IoT about the job status and receives notifications
45 * and datablock over `job` and `stream` topics.
46 *
47 * The OTA MQTT interface is defined in @ref ota_mqtt_interface.h.
48 * <br>
49 *
50 * The functions that must be implemented are:<br>
51 * - [OTA MQTT Subscribe](@ref OtaMqttSubscribe_t)
52 * - [OTA MQTT Unsubscribe](@ref OtaMqttSubscribe_t)
53 * - [OTA MQTT Publish](@ref OtaMqttSubscribe_t)
54 *
55 * These functions can be grouped into the structure `OtaMqttInterface_t`
56 * and passed to @ref OtaInterfaces_t to represent the MQTT interface.
57 * @code{c}
58 * OtaMqttInterface_t mqttInterface;
59 * mqttInterface.subscribe = mqttSubscribe;
60 * mqttInterface.unsubscribe = mqttUnsubscribe;
61 * mqttInterface.publish = mqttPublish;
62 *
63 * ....
64 *
65 * OtaInterfaces_t otaInterfaces;
66 * otaInterfaces.mqtt = mqttInterface
67 *
68 * ....
69 *
70 * OTA_Init( &otaBuffer,
71 * &otaInterfaces,
72 * ( CLIENT_IDENTIFIER ),
73 * otaAppCallback )
74 * @endcode
75 */
76
77 /**
78 * @ingroup ota_enum_types
79 * @brief The OTA MQTT interface return status.
80 */
81 typedef enum OtaMqttStatus
82 {
83 OtaMqttSuccess = 0, /*!< @brief OTA MQTT interface success. */
84 OtaMqttPublishFailed = 0xa0, /*!< @brief Attempt to publish a MQTT message failed. */
85 OtaMqttSubscribeFailed, /*!< @brief Failed to subscribe to a topic. */
86 OtaMqttUnsubscribeFailed /*!< @brief Failed to unsubscribe from a topic. */
87 } OtaMqttStatus_t;
88
89 /**
90 * @brief Subscribe to the Mqtt topics.
91 *
92 * This function subscribes to the Mqtt topics with the Quality of service
93 * received as parameter. This function also registers a callback for the
94 * topicfilter.
95 *
96 * @param[pTopicFilter] Mqtt topic filter.
97 *
98 * @param[topicFilterLength] Length of the topic filter.
99 *
100 * @param[ucQoS] Quality of Service
101 *
102 * @return OtaMqttSuccess if success , other error code on failure.
103 */
104
105 typedef OtaMqttStatus_t ( * OtaMqttSubscribe_t ) ( const char * pTopicFilter,
106 uint16_t topicFilterLength,
107 uint8_t ucQoS );
108
109 /**
110 * @brief Unsubscribe to the Mqtt topics.
111 *
112 * This function unsubscribes to the Mqtt topics with the Quality of service
113 * received as parameter.
114 *
115 * @param[pTopicFilter] Mqtt topic filter.
116 *
117 * @param[topicFilterLength] Length of the topic filter.
118 *
119 * @param[ucQoS] Quality of Service
120 *
121 * @return OtaMqttSuccess if success , other error code on failure.
122 */
123
124 typedef OtaMqttStatus_t ( * OtaMqttUnsubscribe_t ) ( const char * pTopicFilter,
125 uint16_t topicFilterLength,
126 uint8_t ucQoS );
127
128 /**
129 * @brief Publish message to a topic.
130 *
131 * This function publishes a message to a given topic & QoS.
132 *
133 * @param[pacTopic] Mqtt topic filter.
134 *
135 * @param[usTopicLen] Length of the topic filter.
136 *
137 * @param[pcMsg] Message to publish.
138 *
139 * @param[ulMsgSize] Message size.
140 *
141 * @param[ucQoS] Quality of Service
142 *
143 * @return OtaMqttSuccess if success , other error code on failure.
144 */
145 typedef OtaMqttStatus_t ( * OtaMqttPublish_t )( const char * const pacTopic,
146 uint16_t usTopicLen,
147 const char * pcMsg,
148 uint32_t ulMsgSize,
149 uint8_t ucQoS );
150
151 /**
152 * @ingroup ota_struct_types
153 * @brief OTA Event Interface structure.
154 */
155 typedef struct OtaMqttInterface
156 {
157 OtaMqttSubscribe_t subscribe; /*!< @brief Interface for subscribing to Mqtt topics. */
158 OtaMqttUnsubscribe_t unsubscribe; /*!< @brief interface for unsubscribing to MQTT topics. */
159 OtaMqttPublish_t publish; /*!< @brief Interface for publishing MQTT messages. */
160 } OtaMqttInterface_t;
161
162 #endif /* ifndef OTA_MQTT_INTERFACE_H */
