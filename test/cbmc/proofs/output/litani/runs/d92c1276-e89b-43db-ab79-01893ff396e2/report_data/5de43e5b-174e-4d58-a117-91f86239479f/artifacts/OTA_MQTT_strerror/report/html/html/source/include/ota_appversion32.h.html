23 /**
24 * @file ota_appversion32.h
25 * @brief Structure to represent the application build version.
26 */
27
28 #ifndef IOT_APPVERSION32_H
29 #define IOT_APPVERSION32_H
30
31 /* Standard includes. */
32 #include <stdint.h>
33
34 /**
35 * @ingroup ota_struct_types
36 * @brief Application version structure.
37 *
38 */
39 typedef struct
40 {
41 union
42 {
43 #if ( defined( __BYTE_ORDER__ ) && defined( __ORDER_LITTLE_ENDIAN__ ) && __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__ ) || ( __little_endian__ == 1 ) || WIN32 || ( __BYTE_ORDER == __LITTLE_ENDIAN )
44 struct version
45 {
46 uint16_t build; /*!< @brief Build of the firmware (Z in firmware version Z.Y.X). */
47 uint8_t minor; /*!< @brief Minor version number of the firmware (Y in firmware version Z.Y.X). */
48
49 uint8_t major; /*!< @brief Major version number of the firmware (X in firmware version Z.Y.X). */
50 } x; /*!< @brief Version number of the firmware. */
51 #elif ( defined( __BYTE_ORDER__ ) && defined( __ORDER_BIG_ENDIAN__ ) && __BYTE_ORDER__ == __ORDER_BIG_ENDIAN__ ) || ( __big_endian__ == 1 ) || ( __BYTE_ORDER == __BIG_ENDIAN )
52 struct version
53 {
54 uint8_t major; /*!< @brief Major version number of the firmware (X in firmware version X.Y.Z). */
55 uint8_t minor; /*!< @brief Minor version number of the firmware (Y in firmware version X.Y.Z). */
56
57 uint16_t build; /*!< @brief Build of the firmware (Z in firmware version X.Y.Z). */
58 } x; /*!< @brief Version number of the firmware. */
59 #else /* if ( defined( __BYTE_ORDER__ ) && defined( __ORDER_LITTLE_ENDIAN__ ) && __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__ ) || ( __little_endian__ == 1 ) || WIN32 || ( __BYTE_ORDER == __LITTLE_ENDIAN ) */
60 #error "Unable to determine byte order!"
61 #endif /* if ( defined( __BYTE_ORDER__ ) && defined( __ORDER_LITTLE_ENDIAN__ ) && __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__ ) || ( __little_endian__ == 1 ) || WIN32 || ( __BYTE_ORDER == __LITTLE_ENDIAN ) */
62 uint32_t unsignedVersion32;
63 int32_t signedVersion32;
64 } u; /*!< @brief Version based on configuration in big endian or little endian. */
65 } AppVersion32_t;
66
67 extern const AppVersion32_t appFirmwareVersion; /*!< @brief Making the version number available globally through external linkage. */
68
69 #endif /* ifndef IOT_APPVERSION32_H */
