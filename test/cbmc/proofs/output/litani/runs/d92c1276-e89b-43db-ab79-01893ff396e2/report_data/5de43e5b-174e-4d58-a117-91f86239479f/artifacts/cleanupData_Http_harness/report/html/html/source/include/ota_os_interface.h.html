33 /**
34 * @otaosfipage
35 * @brief The OTA OS Functional Interface definition.
36 *
37 * @otaosfisectionoverview
38 *
39 * The OTA OS Functional interface is a set of APIs that must be implemented
40 * for the device using the OTA library. The function implementations for this
41 * interface are provided to the OTA library in the user application. The OTA
42 * library calls the function implementations to perform functionalities that
43 * are typically provided by an operating system. This includes managing
44 * events, timers, and memory allocation. While these are typically provided by
45 * operating systems, an operating system is not required. Any implementation
46 * of these functionalities that meet the requirements of the interface will
47 * work.
48 *
49 * The OTA OS Functional Interface is defined in @ref ota_os_interface.h.
50 * <br>
51 *
52 * The functions that must be implemented are:<br>
53 * - [OTA OS Functional Interface Initialize Event Mechanism](@ref OtaInitEvent_t)
54 * - [OTA OS Functional Interface Send Event](@ref OtaSendEvent_t)
55 * - [OTA OS Functional Interface Receive Event](@ref OtaReceiveEvent_t)
56 * - [OTA OS Functional Interface Deinitialize Event](@ref OtaDeinitEvent_t)
57 * - [OTA OS Functional Interface Timer Callback](@ref OtaTimerCallback_t)
58 * - [OTA OS Functional Interface Start Timer](@ref OtaStartTimer_t)
59 * - [OTA OS Functional Interface Stop Timer](@ref OtaStopTimer_t)
60 * - [OTA OS Functional Interface Delete Timer](@ref OtaDeleteTimer_t)
61 * - [OTA OS Functional Interface Malloc](@ref OtaMalloc_t)
62 * - [OTA OS Functional Interface Free](@ref OtaFree_t)
63 *
64 * An example implementation for the OTA OS Functional Interface for FreeRTOS
65 * can be found in the files ota_os_freertos.c and ota_os_freertos.h.
66 *
67 * An example implementation for the OTA OS Functional Interface for POSIX can
68 * be found in the files ota_os_posix.c and ota_os_posix.h.
69 */
70

88 /**
89 * @ingroup ota_enum_types
90 * @brief The OTA OS interface return status.
91 */
92 typedef enum OtaOsStatus
93 {
94 OtaOsSuccess = 0, /*!< @brief OTA OS interface success. */
95 OtaOsEventQueueCreateFailed = 0x80U, /*!< @brief Failed to create the event queue. */
96 OtaOsEventQueueSendFailed, /*!< @brief Posting event message to the event queue failed. */
97 OtaOsEventQueueReceiveFailed, /*!< @brief Failed to receive from the event queue. */
98 OtaOsEventQueueDeleteFailed, /*!< @brief Failed to delete the event queue. */
99 OtaOsTimerCreateFailed, /*!< @brief Failed to create the timer. */
100 OtaOsTimerStartFailed, /*!< @brief Failed to create the timer. */
101 OtaOsTimerRestartFailed, /*!< @brief Failed to restart the timer. */
102 OtaOsTimerStopFailed, /*!< @brief Failed to stop the timer. */
103 OtaOsTimerDeleteFailed /*!< @brief Failed to delete the timer. */
104 } OtaOsStatus_t;
105
106 /**
107 * @brief Initialize the OTA events.
108 *
109 * This function initializes the OTA events mechanism.
110 *
111 * @param[pEventCtx] Pointer to the OTA event context.
112 *
113 * @return OtaOsStatus_t, OtaOsSuccess if success , other error code on failure.
114 */
115
116 typedef OtaOsStatus_t ( * OtaInitEvent_t ) ( OtaEventContext_t * pEventCtx );
117
118 /**
119 * @brief Sends an OTA event.
120 *
121 * This function sends an event to OTA library event handler.
122 *
123 * @param[pEventCtx] Pointer to the OTA event context.
124 *
125 * @param[pEventMsg] Event to be sent to the OTA handler.
126 *
127 * @param[timeout] The maximum amount of time (msec) the task should block.
128 *
129 * @return OtaOsStatus_t, OtaOsSuccess if success , other error code on failure.
130 */
131
132 typedef OtaOsStatus_t ( * OtaSendEvent_t )( OtaEventContext_t * pEventCtx,
133 const void * pEventMsg,
134 unsigned int timeout );
135
136 /**
137 * @brief Receive an OTA event.
138 *
139 * This function receives next event from the pending OTA events.
140 *
141 * @param[pEventCtx] Pointer to the OTA event context.
142 *
143 * @param[pEventMsg] Pointer to store message.
144 *
145 * @param[timeout] The maximum amount of time the task should block.
146 *
147 * @return OtaOsStatus_t, OtaOsSuccess if success , other error code on failure.
148 */
149
150 typedef OtaOsStatus_t ( * OtaReceiveEvent_t )( OtaEventContext_t * pEventCtx,
151 void * pEventMsg,
152 uint32_t timeout );
153
154 /**
155 * @brief Deinitialize the OTA Events mechanism.
156 *
157 * This function deinitialize the OTA events mechanism and frees any resources
158 * used.
159 *
160 * @param[pEventCtx] Pointer to the OTA event context.
161 *
162 * @return OtaOsStatus_t, OtaOsSuccess if success , other error code on failure.
163 */
164
165 typedef OtaOsStatus_t ( * OtaDeinitEvent_t )( OtaEventContext_t * pEventCtx );
166
167 /**
168 * @brief Timer callback.
169 *
170 * Type definition for timer callback.
171 *
172 * @param[otaTimerId] Timer ID of type otaTimerId_t
173 *
174 * @return OtaOsStatus_t, OtaOsSuccess if success , other error code on failure.
175 */
176
177 typedef void ( * OtaTimerCallback_t )( OtaTimerId_t otaTimerId );
178
179 /**
180 * @brief Start timer.
181 *
182 * This function starts the timer or resets it if it has already started.
183 *
184 * @param[otaTimerId] Timer ID of type otaTimerId_t
185 *
186 * @param[pTimerName] Timer name.
187 *
188 * @param[timeout] Timeout for the timer in milliseconds.
189 *
190 * @param[callback] Callback to be called when timer expires.
191 *
192 * @return OtaOsStatus_t, OtaOsSuccess if success , other error code on failure.
193 */
194
195 typedef OtaOsStatus_t ( * OtaStartTimer_t ) ( OtaTimerId_t otaTimerId,
196 const char * const pTimerName,
197 const uint32_t timeout,
198 OtaTimerCallback_t callback );
199
200 /**
201 * @brief Stop timer.
202 *
203 * This function stops the time.
204 *
205 * @param[otaTimerId] Timer ID of type otaTimerId_t
206 *
207 * @return OtaOsStatus_t, OtaOsSuccess if success , other error code on failure.
208 */
209
210 typedef OtaOsStatus_t ( * OtaStopTimer_t ) ( OtaTimerId_t otaTimerId );
211
212 /**
213 * @brief Delete a timer.
214 *
215 * This function deletes a timer.
216 *
217 * @param[otaTimerId] Timer ID of type otaTimerId_t
218 *
219 * @return OtaOsStatus_t, OtaOsSuccess if success , other error code on failure.
220 */
221
222 typedef OtaOsStatus_t ( * OtaDeleteTimer_t ) ( OtaTimerId_t otaTimerId );
223
224 /**
225 * @brief Allocate memory.
226 *
227 * This function allocates the requested memory and returns a pointer to it.
228 *
229 * @param[size] This is the size of the memory block, in bytes..
230 *
231 * @return This function returns a pointer to the allocated memory, or NULL if
232 * the request fails.
233 */
234
235 typedef void * ( * OtaMalloc_t ) ( size_t size );
236
237 /**
238 * @brief Free memory.
239 *
240 * This function deallocates the memory previously allocated by a call to allocation
241 * function of type OtaMalloc_t.
242 *
243 * @param[ptr] This is the pointer to a memory block previously allocated with function
244 * of type OtaMalloc_t. If a null pointer is passed as argument, no action occurs.
245 *
246 * @return None.
247 */
248
249 typedef void ( * OtaFree_t ) ( void * ptr );

275 /**
276 * @ingroup ota_struct_types
277 * @brief OTA memory allocation interface.
278 */
279 typedef struct OtaMallocInterface
280 {
281 /* MISRA rule 21.3 prohibits the use of malloc and free from stdlib.h, however, we're only
282 * defining the interface here. On FreeRTOS this is implemented with pvPortMalloc and vPortFree,
283 * and on Linux it's implemented with standard C malloc and free. This is a false positive. */
284 /* coverity[misra_c_2012_rule_21_3_violation] */
285 OtaMalloc_t malloc; /*!< @brief OTA memory allocate interface. */
286 /* coverity[misra_c_2012_rule_21_3_violation] */
287 OtaFree_t free; /*!< @brief OTA memory deallocate interface. */
288 } OtaMallocInterface_t;

299 } OtaOSInterface_t;
