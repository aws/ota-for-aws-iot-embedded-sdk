23 /**
24 * @file ota_http_interface.h
25 * @brief Contains OTA HTTP Statuses, function type definitions and http interface structure.
26 */
27
28 #ifndef OTA_HTTP_INTERFACE_H
29 #define OTA_HTTP_INTERFACE_H
30
31 /* Standard library includes. */
32 #include <stddef.h>
33 #include <stdint.h>
34
35 /**
36 * @otahttppage
37 * @brief The OTA PAL interface definition.
38 *
39 * @otahttpsectionoverview
40 *
41 * The OTA MQTT interface is a set of APIs that must be implemented by
42 * a library to enable the OTA library to download a file block by
43 * connecting to a pre-signed url and fetching data blocks.
44 *
45 * The OTA MQTT interface is defined in @ref ota_mqtt_interface.h.
46 * <br>
47 *
48 * The functions that must be implemented are:<br>
49 * - [OTA MQTT Subscribe](@ref OtaMqttSubscribe_t)
50 * - [OTA MQTT Unsubscribe](@ref OtaMqttSubscribe_t)
51 * - [OTA MQTT Publish](@ref OtaMqttSubscribe_t)
52 *
53 * These functions can be grouped into the structure `OtaHttpInterface_t`
54 * and passed to @ref OtaInterfaces_t to represent the MQTT interface.
55 * @code{c}
56 * OtaHttpInterface_t httpInterface;
57 * httpInterface.init = httpInit;
58 * httpInterface.request = httpRequest;
59 * httpInterface.deinit = httpDeinit;
60 *
61 * .....
62 *
63 * OtaInterfaces_t otaInterfaces;
64 * otaInterfaces.http = httpInterface
65 *
66 * .....
67 *
68 * OTA_Init( &otaBuffer,
69 * &otaInterfaces,
70 * ( CLIENT_IDENTIFIER ),
71 * otaAppCallback )
72 * @endcode
73 */
74
75 /**
76 * @ingroup ota_enum_types
77 * @brief The OTA HTTP interface return status.
78 */
79 typedef enum OtaHttpStatus
80 {
81 OtaHttpSuccess = 0, /*!< @brief OTA HTTP interface success. */
82 OtaHttpInitFailed = 0xc0, /*!< @brief Error initializing the HTTP connection. */
83 OtaHttpDeinitFailed, /*!< @brief Error deinitializing the HTTP connection. */
84 OtaHttpRequestFailed /*!< @brief Error sending the HTTP request. */
85 } OtaHttpStatus_t;
86
87 /**
88 * @brief Init OTA Http interface.
89 *
90 * This function parses the pre-signed url and initializes connection.
91 *
92 * @param[in] pUrl Pointer to the pre-signed url for downloading update file.
93 *
94 * @return OtaHttpSuccess if success , other error code on failure.
95 */
96
97 typedef OtaHttpStatus_t ( * OtaHttpInit_t ) ( char * pUrl );
98
99 /**
100 * @brief Request file block over Http.
101 *
102 * This function requests file block over Http from the rangeStart and rangeEnd.
103 *
104 * @param[in] rangeStart Starting index of the file data to be requested.
105 *
106 * @param[in] rangeEnd End index of the file data to be requested.
107 *
108 * @return OtaHttpSuccess if success , other error code on failure.
109 */
110
111 typedef OtaHttpStatus_t ( * OtaHttpRequest_t ) ( uint32_t rangeStart,
112 uint32_t rangeEnd );
113
114 /**
115 * @brief Deinit OTA Http interface.
116 *
117 * This function cleanups Http connection and other data used for
118 * requesting file blocks using the pre-signed url.
119 *
120 * @return OtaHttpSuccess if success , other error code on failure.
121 */
122 typedef OtaHttpStatus_t ( * OtaHttpDeinit )( void );
123
124 /**
125 * @ingroup ota_struct_types
126 * @brief OTA Event Interface structure.
127 *
128 */
129 typedef struct OtaHttpInterface
130 {
131 OtaHttpInit_t init; /*!< @brief Reference to HTTP initialization. */
132 OtaHttpRequest_t request; /*!< @brief Reference to HTTP data request. */
133 OtaHttpDeinit deinit; /*!< @brief Reference to HTTP deinitialize. */
134 } OtaHttpInterface_t;
135
136 #endif /* ifndef OTA_HTTP_INTERFACE_H */
