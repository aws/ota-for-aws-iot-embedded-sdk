23 /**
24 * @file ota_http_private.h
25 * @brief Contains definitions of routines for OTA download using HTTP data plane.
26 */
27
28 #ifndef OTA_HTTP_H
29 #define OTA_HTTP_H
30
31 /* OTA includes. */
32 #include "ota.h"
33 #include "ota_private.h"
34
35
36 /**
37 * @brief Initialize file transfer over HTTP.
38 *
39 * This function initializes the file transfer after the OTA job is parsed and accepted
40 * by initializing the http component with pre-signed url.
41 *
42 * @param[in] pAgentCtx The OTA agent context.
43 *
44 * @return The OTA error code. See OTA Agent error codes information in ota.h.
45 */
46 OtaErr_t initFileTransfer_Http( OtaAgentContext_t * pAgentCtx );
47
48
49 /**
50 * @brief Request File block over HTTP.
51 *
52 * This function is used for requesting a file block over HTTP using the
53 * file context.
54 *
55 * @param[in] pAgentCtx The OTA agent context.
56 *
57 * @return The OTA PAL layer error code combined with the MCU specific error code. See OTA Agent
58 * error codes information in ota.h.
59 */
60 OtaErr_t requestDataBlock_Http( OtaAgentContext_t * pAgentCtx );
61
62
63 /**
64 * @brief Stub for decoding the file block.
65 *
66 * File block received over HTTP does not require decoding, only increment the number
67 * of blocks received.
68 *
69 * @param[in] pMessageBuffer The message to be decoded.
70 * @param[in] messageSize The size of the message in bytes.
71 * @param[out] pFileId The server file ID.
72 * @param[out] pBlockId The file block ID.
73 * @param[out] pBlockSize The file block size.
74 * @param[out] pPayload The payload.
75 * @param[out] pPayloadSize The payload size.
76 *
77 * @return The OTA PAL layer error code combined with the MCU specific error code. See OTA Agent
78 * error codes information in ota.h.
79 */
80 OtaErr_t decodeFileBlock_Http( const uint8_t * pMessageBuffer,
81 size_t messageSize,
82 int32_t * pFileId,
83 int32_t * pBlockId,
84 int32_t * pBlockSize,
85 uint8_t ** pPayload,
86 size_t * pPayloadSize );
87
88 /**
89 * @brief Cleanup related to OTA data plane over HTTP.
90 *
91 * This function performs cleanup by resetting the number of blocks received
92 * and deinit the http component.
93 *
94 * @param[in] pAgentCtx The OTA agent context.
95 *
96 * @return The OTA error code. See OTA Agent error codes information in ota.h.
97 */
98
99 OtaErr_t cleanupData_Http( const OtaAgentContext_t * pAgentCtx );
100
101 /**
102 * @brief Status to string conversion for OTA HTTP interface status.
103 *
104 * @param[in] status The status to convert to a string.
105 *
106 * @return The string representation of the status.
107 */
108 const char * OTA_HTTP_strerror( OtaHttpStatus_t status );
109
110 #endif /* ifndef OTA_HTTP_H */
