23 /**
24 * @file decodeFileBlock_Http_harness.c
25 * @brief Implements the proof harness for decodeFileBlock_Http function.
26 */
27 /* Http interface includes. */
28 #include "ota_http_private.h"
29
30
31 void decodeFileBlock_Http_harness()
32 {
33 OtaErr_t err;
34
35 uint8_t * pMessageBuffer;
36 size_t messageSize;
37 int32_t * pFileId;
38 int32_t * pBlockId;
39 int32_t * pBlockSize;
40 uint8_t ** pPayload;
41 size_t * pPayloadSize;
42
43 int32_t fileId;
44 int32_t blockId;
45 int32_t blockSize;
46 size_t payloadSize;
47
48 /* Initializing the unconstrained variables. */
49 pFileId = &fileId;
50 pBlockId = &blockId;
51 pBlockSize = &blockSize;
52
53 /* Allocate memory for the payload and the buffer to store the message. */
54 char * size = ( char * ) malloc( messageSize );
55 pMessageBuffer = ( uint8_t * ) malloc( messageSize );
56
57 /* size is used to store the content from the message buffer and should
58 * not point to a NULL address. */
59 __CPROVER_assume( size != NULL );
60
61 /* Initializing the variable. */
62 pPayload = &size;
63 pPayloadSize = &payloadSize;
64
65 /* pMessageBuffer stores the message from the FIle and should point to a NULL
66 * address. */
67 __CPROVER_assume( pMessageBuffer != NULL );
68
69 /* pFileId is required to point to a FILE and cannot point to a NULL address. */
70 __CPROVER_assume( pFileId != NULL );
71
72 /* BlockId is pointer to the block inside the file and should not point to a NULL
73 * address. */
74 __CPROVER_assume( pBlockId != NULL );
75
76 /* pBlockSize is a pointer to store the size of the current block and should not
77 * point to a NULL address. */
78 __CPROVER_assume( pBlockSize != NULL );
79
80 /* pPayloadSize is the pointer to a variable which stores the size of the Payload
81 * and should not be point to NULL address. */
82 __CPROVER_assume( pPayloadSize != NULL );
83
84 /* Call the function under test. */
85 err = decodeFileBlock_Http( pMessageBuffer, messageSize, pFileId,
86 pBlockId, pBlockSize, pPayload, pPayloadSize );
87
88 /* Assert because the function cannot return values other OtaErrNone and OtaErrInvalidArg. */
89 __CPROVER_assert( ( ( err == OtaErrNone ) || ( err == OtaErrInvalidArg ) ),
90 "Function return should be either None or invalid argument" );
91
92 /* Free allocated memory. */
93 free( size );
94 free( pMessageBuffer );
95 }
