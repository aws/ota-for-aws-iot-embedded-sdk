23 /**
24 * @file cleanupData_Http_harness.c
25 * @brief Implements the proof harness for cleanupData_Http function.
26 */
27 /* Http Interface includes. */
28 #include "ota_http_private.h"
29
30 /* Stub required for the proof. */
31 OtaHttpStatus_t deinit()
32 {
33 OtaHttpStatus_t status;
34
35 return status;
36 }
37 /*-----------------------------------------------------------*/
38
39 void cleanupData_Http_harness()
40 {
41 OtaAgentContext_t * pAgentCtx;
42 OtaHttpInterface_t pHttp;
43 OtaInterfaces_t * pInterfaces;
44 OtaHttpStatus_t status;
45
46 pAgentCtx = ( OtaAgentContext_t * ) malloc( sizeof( OtaAgentContext_t ) );
47 pInterfaces = ( OtaInterfaces_t * ) malloc( sizeof( OtaInterfaces_t ) );
48
49 /* The function requires Agent and the interface to be initialized and thus
50 * they can't be NULL. If they are, they will hit an assert in the function. */
51 __CPROVER_assume( pAgentCtx != NULL && pInterfaces != NULL );
52
53 /* Updating the function pointer in pHttp to the stub. */
54 pHttp.deinit = deinit;
55
56 /* Update the interface and the Agent. */
57 pInterfaces->http = pHttp;
58 pAgentCtx->pOtaInterface = pInterfaces;
59
60 /* Call the function under test. */
61 status = cleanupData_Http( pAgentCtx );
62
63 /* The function should either return OtaErrNone or OtaErrCleanupDataFailed. */
64 __CPROVER_assert( ( status == OtaErrNone ) ||
65 ( status == OtaErrCleanupDataFailed ),
66 "The function return should be either OtaErrNone or OtaErrCleanupDataFailed" );
67 }
