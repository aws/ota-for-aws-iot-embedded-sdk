25 * @brief Implements the proof harness for requestDataBlock_Http function.
26 */
27 /* Http interface includes. */
28 #include "ota_http_private.h"
29
30 /* Stub required for the proof. */
31 OtaHttpStatus_t request( uint32_t rangeStart,
32 uint32_t rangeEnd )
33 {
34 OtaHttpStatus_t status;
35
36 return status;
37 }
38 /*-----------------------------------------------------------*/
39
40 void requestDataBlock_Http_harness()
41 {
42 OtaAgentContext_t * pAgentCtx;
43 OtaFileContext_t fileContext;
44 OtaHttpInterface_t http;
45 OtaInterfaces_t interface;
46 OtaErr_t err;
47 OtaAgentContext_t agent;
48
49 /* Allocating memory to the agent context. The agent can never be NULL since it
50 * is a globally declared variable. */
51 pAgentCtx = &agent;
52
53 /* Initialize the file context field in the Agent context. */
54 pAgentCtx->fileContext = fileContext;
55 http.request = request;
56 interface.http = http;
57
58 /* File size can never be zero and is verified in the validateAndStartJob() function.*/
59 __CPROVER_assume( pAgentCtx->fileContext.fileSize != 0 );
60
61 /* Initialize the interface in the Agent Context. */
62 pAgentCtx->pOtaInterface = &interface;
63
64 err = requestDataBlock_Http( pAgentCtx );
65
66 /*Assert to check if the return from the function is of expected values. */
67 __CPROVER_assert( ( err == OtaErrNone ) || ( err == OtaErrRequestFileBlockFailed ),
68 "The function return should be either of these values" );
69 }
