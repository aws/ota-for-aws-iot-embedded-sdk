23 /**
24 * @file decodeFileBlock_Http_harness.c
25 * @brief Implements the proof harness for decodeFileBlock_Http function.
26 */
27 /* Http interface includes. */
28 #include "ota_http_private.h"
29
30
31 void decodeFileBlock_Http_harness()
32 {
33 OtaErr_t err;
34
35 uint8_t * pMessageBuffer;
36 size_t messageSize;
37 int32_t * pFileId;
38 int32_t * pBlockId;
39 int32_t * pBlockSize;
40 uint8_t ** pPayload;
41 size_t * pPayloadSize;
42
43 int32_t fileId;
44 int32_t blockId;
45 int32_t blockSize;
46 size_t payloadSize;
47
48 /* Initializing the unconstrained variables. These variables can never be NUll
49 * since they have been declared statically in the decodeAndStoreDataBlock() function. */
50 pFileId = &fileId;
51 pBlockId = &blockId;
52 pBlockSize = &blockSize;
53
54 /* Allocate memory for the payload and the buffer to store the message. */
55 char * size = ( char * ) malloc( messageSize );
56 pMessageBuffer = ( uint8_t * ) malloc( messageSize );
57
58 /* size can never be NULL since it is a pointer to the payload buffer which is initialized
59 * statically in the ingestDataBlock() Function. */
60 __CPROVER_assume( size != NULL );
61
62 /* Initializing the variable. */
63 pPayload = &size;
64 pPayloadSize = &payloadSize;
65
66 /* This assumption is made because the pMessageBuffer is always pointing to a
67 * static array. */
68 __CPROVER_assume( pMessageBuffer != NULL );
69
70 err = decodeFileBlock_Http( pMessageBuffer, messageSize, pFileId,
71 pBlockId, pBlockSize, pPayload, pPayloadSize );
72
73 /* Assert because the function cannot return values other OtaErrNone and OtaErrInvalidArg. */
74 __CPROVER_assert( ( ( err == OtaErrNone ) || ( err == OtaErrInvalidArg ) ),
75 "Function return should be either None or invalid argument" );
76
77 /* Free allocated memory. */
78 free( size );
79 free( pMessageBuffer );
80 }
