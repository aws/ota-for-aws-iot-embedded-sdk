25 * @brief Implements the proof harness for OTA_MQTT_strerror function.

28 #include "ota_mqtt_private.h"
29
30 void OTA_MQTT_strerror_harness()
31 {
32 OtaMqttStatus_t status;
33 const char * pstr;
34
35 pstr = OTA_MQTT_strerror( status );
36
37 /* The function returns a string which is never NULL. If it does, then it is a
38 * problem. */
39 __CPROVER_assert( pstr != NULL, "Function return cannot be NULL" );
40 }
