23 /**
24 * @file ota_interface_private.h
25 * @brief Contains function definitions and structures for data and control interfaces.
26 */
27
28 #ifndef OTA_INTERFACE_PRIVATE_H
29 #define OTA_INTERFACE_PRIVATE_H
30
31 /* OTA includes. */
32 #include "ota.h"
33 #include "ota_private.h"
34
35 /* General Constants. */
36
37 /* OTA control protocol constants. */
38 #define OTA_CONTROL_OVER_MQTT 0x00000001 /*!< Specify control over mqtt. */
39
40 /* OTA data protocol constants. */
41 #define OTA_DATA_OVER_MQTT 0x00000001 /*!< Specify data over mqtt. */
42 #define OTA_DATA_OVER_HTTP 0x00000002 /*!< Specify data over http. */
43 #define OTA_DATA_NUM_PROTOCOLS ( 2U ) /*!< Number of protocols supported. */
44
45
46 /**
47 * @brief Represents the OTA control interface functions.
48 *
49 * The functions in this structure are used for the control operations
50 * during over the air updates like OTA job status updates.
51 */
52 typedef struct
53 {
54 OtaErr_t ( * requestJob )( OtaAgentContext_t * pAgentCtx ); /*!< Request for the next available OTA job from the job service. */
55 OtaErr_t ( * updateJobStatus )( OtaAgentContext_t * pAgentCtx,
56 OtaJobStatus_t status,
57 int32_t reason,
58 int32_t subReason ); /*!< Updates the OTA job status with information like in progress, completion, or failure. */
59 OtaErr_t ( * cleanup )( const OtaAgentContext_t * pAgentCtx ); /*!< Cleanup related to OTA control plane. */
60 } OtaControlInterface_t;
61
62 /**
63 * @brief Represents the OTA data interface functions.
64 *
65 * The functions in this structure are used for the data operations
66 * during over the air updates like requesting file blocks.
67 */
68 typedef struct
69 {
70 OtaErr_t ( * initFileTransfer )( OtaAgentContext_t * pAgentCtx ); /*!< Initialize file transfer. */
71 OtaErr_t ( * requestFileBlock )( OtaAgentContext_t * pAgentCtx ); /*!< Request File block. */
72 OtaErr_t ( * decodeFileBlock )( const uint8_t * pMessageBuffer,
73 size_t messageSize,
74 int32_t * pFileId,
75 int32_t * pBlockId,
76 int32_t * pBlockSize,
77 uint8_t ** pPayload,
78 size_t * pPayloadSize ); /*!< Decode a cbor encoded fileblock. */
79 OtaErr_t ( * cleanup )( const OtaAgentContext_t * pAgentCtx ); /*!< Cleanup related to OTA data plane. */
80 } OtaDataInterface_t;
81
82 /**
83 * @brief Set control interface for OTA operations.
84 *
85 * This function updates the OTA control operation functions as per the config
86 * options selected.
87 *
88 * @param[out] pControlInterface OTA Control interface.
89 *
90 */
91 void setControlInterface( OtaControlInterface_t * pControlInterface );
92
93 /**
94 * @brief Set the data interface used for OTA operations.
95 *
96 * This function updates the OTA data operation based on the config options.
97 * The interface can be set to the MQTT interface or the HTTP interface.
98 *
99 * These interfaces can be enabled with the configENABLED_DATA_PROTOCOLS macro.
100 * The protocol interface that should be prioritized when both protocols are
101 * valid options is configured with the configOTA_PRIMARY_DATA_PROTOCOL macro.
102 *
103 * @param[out] pDataInterface OTA data interface to overwrite.
104 *
105 * @param[in] pProtocol String containing a list of protocols that may be set.
106 *
107 */
108 OtaErr_t setDataInterface( OtaDataInterface_t * pDataInterface,
109 const uint8_t * pProtocol );
110
111 #endif /* ifndef OTA_INTERFACE_PRIVATE_H */
