28 #ifndef OTA_CBOR_H
29 #define OTA_CBOR_H

31 /**
32 * Message field definitions, per the server specification. These are
33 * not part of the library interface but are included here for testability.
34 */
35
36 #define OTA_CBOR_CLIENTTOKEN_KEY "c" /*!< Key for client id. */
37 #define OTA_CBOR_FILEID_KEY "f" /*!< Key for file id. */
38 #define OTA_CBOR_BLOCKSIZE_KEY "l" /*!< Key for file block size. */
39 #define OTA_CBOR_BLOCKOFFSET_KEY "o" /*!< Key for file block offset. */
40 #define OTA_CBOR_BLOCKBITMAP_KEY "b" /*!< Key for bitmap. */
41 #define OTA_CBOR_STREAMDESCRIPTION_KEY "d" /*!< Key for stream name. */
42 #define OTA_CBOR_STREAMFILES_KEY "r" /*!< Key for file attributes. */
43 #define OTA_CBOR_FILESIZE_KEY "z" /*!< Key for file size. */
44 #define OTA_CBOR_BLOCKID_KEY "i" /*!< Key for block id. */
45 #define OTA_CBOR_BLOCKPAYLOAD_KEY "p" /*!< Key for payload of a block. */
46 #define OTA_CBOR_NUMBEROFBLOCKS_KEY "n" /*!< Key for number of blocks. */
47
48 /**
49 * @brief Decode a Get Stream response message from AWS IoT OTA.
50 */
51 bool OTA_CBOR_Decode_GetStreamResponseMessage( const uint8_t * pMessageBuffer,
52 size_t messageSize,
53 int32_t * pFileId,
54 int32_t * pBlockId,
55 int32_t * pBlockSize,
56 uint8_t ** pPayload,
57 size_t * pPayloadSize );
58
59 /**
60 * @brief Create an encoded Get Stream Request message for the AWS IoT OTA
61 * service. The service allows block count or block bitmap to be requested,
62 * but not both.
63 */
64 bool OTA_CBOR_Encode_GetStreamRequestMessage( uint8_t * pMessageBuffer,
65 size_t messageBufferSize,
66 size_t * pEncodedMessageSize,
67 const char * pClientToken,
68 int32_t fileId,
69 int32_t blockSize,
70 int32_t blockOffset,
71 uint8_t * pBlockBitmap,
72 size_t blockBitmapSize,
73 int32_t numOfBlocksRequested );
74
75 #endif /* ifndef OTA_CBOR_H */
